Option Explicit

' =========================================================
' source_meetings.xlsm / source_meetings.xlms 用
' 特別委員会系 3シート専用出力
' =========================================================

Public Sub ExportSpecialCommitteeAll()
    On Error GoTo ErrHandler

    ExportSingleSheet "特別委員会マスタ", _
                      "app\data\committees\special_committees.js", _
                      "special_committees"

    ExportSingleSheet "特別委員会設置履歴", _
                      "app\data\committees\special_committee_instances.js", _
                      "special_committee_instances"

    ExportSingleSheet "特別委員会委員名簿履歴", _
                      "app\data\committees\special_committee_members.js", _
                      "special_committee_members"

    MsgBox "特別委員会系データの出力が完了しました。", vbInformation
    Exit Sub

ErrHandler:
    MsgBox "ExportSpecialCommitteeAll エラー: " & Err.Description, vbCritical
End Sub

Public Sub ExportActiveSpecialCommitteeSheet()
    Dim sheetName As String
    Dim outputPath As String
    Dim dataKey As String

    On Error GoTo ErrHandler

    sheetName = ActiveSheet.Name

    Select Case sheetName
        Case "特別委員会マスタ"
            outputPath = "app\data\committees\special_committees.js"
            dataKey = "special_committees"

        Case "特別委員会設置履歴"
            outputPath = "app\data\committees\special_committee_instances.js"
            dataKey = "special_committee_instances"

        Case "特別委員会委員名簿履歴"
            outputPath = "app\data\committees\special_committee_members.js"
            dataKey = "special_committee_members"

        Case Else
            Err.Raise vbObjectError + 6001, , "このシートは出力対象ではありません: " & sheetName
    End Select

    ExportSingleSheet sheetName, outputPath, dataKey

    MsgBox "アクティブシートの出力が完了しました。", vbInformation
    Exit Sub

ErrHandler:
    MsgBox "ExportActiveSpecialCommitteeSheet エラー: " & Err.Description, vbCritical
End Sub

Private Sub ExportSingleSheet(ByVal sheetName As String, ByVal relativePath As String, ByVal dataKey As String)
    Dim ws As Worksheet
    Dim jsText As String
    Dim outputPath As String

    On Error GoTo ErrHandler

    Set ws = ThisWorkbook.Worksheets(sheetName)

    jsText = BuildJsFromSheet(ws, dataKey)
    outputPath = ThisWorkbook.Path & "\" & relativePath

    EnsureParentFolderExists outputPath
    WriteTextFileUtf8 outputPath, jsText

    Debug.Print "Exported: " & sheetName & " -> " & outputPath
    Exit Sub

ErrHandler:
    Err.Raise Err.Number, , "ExportSingleSheet [" & sheetName & "] : " & Err.Description
End Sub

Private Function BuildJsFromSheet(ByVal ws As Worksheet, ByVal dataKey As String) As String
    Dim lastRow As Long
    Dim lastCol As Long
    Dim r As Long
    Dim c As Long
    Dim headers() As String
    Dim sb As String
    Dim rowText As String
    Dim headerName As String
    Dim v As Variant

    lastRow = GetLastDataRow(ws)
    lastCol = GetLastHeaderCol(ws)

    If lastCol = 0 Then
        Err.Raise vbObjectError + 5001, , "1行目にヘッダが見つかりません。"
    End If

    ReDim headers(1 To lastCol)

    For c = 1 To lastCol
        headers(c) = Trim$(CStr(ws.Cells(1, c).Value))
        If headers(c) = "" Then
            Err.Raise vbObjectError + 5002, , "1行目に空ヘッダがあります。列=" & c
        End If
    Next c

    sb = ""
    sb = sb & "window.APP_DATA = window.APP_DATA || {};" & vbCrLf
    sb = sb & "window.APP_DATA." & dataKey & " = [" & vbCrLf

    If lastRow >= 2 Then
        For r = 2 To lastRow
            If Not IsEntireRowEmpty(ws, r, lastCol) Then
                rowText = "  {" & vbCrLf

                For c = 1 To lastCol
                    headerName = headers(c)
                    v = ws.Cells(r, c).Value

                    rowText = rowText & "    " & headerName & ": " & ConvertCellToJsValue(v, headerName)

                    If c < lastCol Then
                        rowText = rowText & ","
                    End If

                    rowText = rowText & vbCrLf
                Next c

                rowText = rowText & "  }"

                If HasNextNonEmptyRow(ws, r, lastRow, lastCol) Then
                    rowText = rowText & ","
                End If

                rowText = rowText & vbCrLf
                sb = sb & rowText
            End If
        Next r
    End If

    sb = sb & "];" & vbCrLf
    BuildJsFromSheet = sb
End Function

Private Function ConvertCellToJsValue(ByVal v As Variant, ByVal headerName As String) As String
    If IsError(v) Then
        ConvertCellToJsValue = """"""
        Exit Function
    End If

    If IsEmpty(v) Then
        ConvertCellToJsValue = """"""
        Exit Function
    End If

    If Trim$(CStr(v)) = "" Then
        ConvertCellToJsValue = """"""
        Exit Function
    End If

    If IsNumericField(headerName) Then
        ConvertCellToJsValue = NormalizeNumericLike(v)
        Exit Function
    End If

    If IsDateField(headerName) Then
        ConvertCellToJsValue = """" & EscapeJsString(FormatDateForJs(v)) & """"
        Exit Function
    End If

    ConvertCellToJsValue = """" & EscapeJsString(CStr(v)) & """"
End Function

Private Function IsNumericField(ByVal headerName As String) As Boolean
    Select Case LCase$(Trim$(headerName))
        Case "sort_order"
            IsNumericField = True
        Case Else
            IsNumericField = False
    End Select
End Function

Private Function IsDateField(ByVal headerName As String) As Boolean
    Dim s As String
    s = LCase$(Trim$(headerName))

    If Right$(s, 5) = "_date" Then
        IsDateField = True
        Exit Function
    End If

    If Right$(s, 3) = "_at" Then
        IsDateField = True
        Exit Function
    End If

    IsDateField = False
End Function

Private Function NormalizeNumericLike(ByVal v As Variant) As String
    If IsNumeric(v) Then
        If CDbl(v) = Int(CDbl(v)) Then
            NormalizeNumericLike = CStr(CLng(v))
        Else
            NormalizeNumericLike = Replace(CStr(CDbl(v)), ",", "")
        End If
    Else
        NormalizeNumericLike = "0"
    End If
End Function

Private Function FormatDateForJs(ByVal v As Variant) As String
    If IsDate(v) Then
        FormatDateForJs = Format$(CDate(v), "yyyy-mm-dd")
    Else
        FormatDateForJs = Trim$(CStr(v))
    End If
End Function

Private Function EscapeJsString(ByVal s As String) As String
    Dim t As String
    t = s
    t = Replace(t, "\", "\\")
    t = Replace(t, """", "\""")
    t = Replace(t, vbCrLf, "\n")
    t = Replace(t, vbCr, "\n")
    t = Replace(t, vbLf, "\n")
    EscapeJsString = t
End Function

Private Sub EnsureParentFolderExists(ByVal filePath As String)
    Dim fso As Object
    Dim parentPath As String

    Set fso = CreateObject("Scripting.FileSystemObject")
    parentPath = fso.GetParentFolderName(filePath)

    If parentPath <> "" Then
        CreateFoldersRecursive parentPath
    End If
End Sub

Private Sub CreateFoldersRecursive(ByVal folderPath As String)
    Dim fso As Object
    Dim parentPath As String

    Set fso = CreateObject("Scripting.FileSystemObject")

    If fso.FolderExists(folderPath) Then Exit Sub

    parentPath = fso.GetParentFolderName(folderPath)
    If parentPath <> "" And Not fso.FolderExists(parentPath) Then
        CreateFoldersRecursive parentPath
    End If

    If Not fso.FolderExists(folderPath) Then
        fso.CreateFolder folderPath
    End If
End Sub

Private Sub WriteTextFileUtf8(ByVal filePath As String, ByVal textContent As String)
    Dim stm As Object

    Set stm = CreateObject("ADODB.Stream")
    stm.Type = 2
    stm.Charset = "utf-8"
    stm.Open
    stm.WriteText textContent
    stm.SaveToFile filePath, 2
    stm.Close
End Sub

Private Function GetLastDataRow(ByVal ws As Worksheet) As Long
    Dim lastCell As Range

    On Error Resume Next
    Set lastCell = ws.Cells.Find(What:="*", After:=ws.Cells(1, 1), LookIn:=xlFormulas, _
                                 LookAt:=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlPrevious)
    On Error GoTo 0

    If lastCell Is Nothing Then
        GetLastDataRow = 0
    Else
        GetLastDataRow = lastCell.Row
    End If
End Function

Private Function GetLastHeaderCol(ByVal ws As Worksheet) As Long
    Dim c As Long
    Dim lastCol As Long

    lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column

    If Trim$(CStr(ws.Cells(1, lastCol).Value)) = "" Then
        GetLastHeaderCol = 0
        Exit Function
    End If

    For c = lastCol To 1 Step -1
        If Trim$(CStr(ws.Cells(1, c).Value)) <> "" Then
            GetLastHeaderCol = c
            Exit Function
        End If
    Next c

    GetLastHeaderCol = 0
End Function

Private Function IsEntireRowEmpty(ByVal ws As Worksheet, ByVal rowNum As Long, ByVal lastCol As Long) As Boolean
    Dim c As Long

    For c = 1 To lastCol
        If Trim$(CStr(ws.Cells(rowNum, c).Value)) <> "" Then
            IsEntireRowEmpty = False
            Exit Function
        End If
    Next c

    IsEntireRowEmpty = True
End Function

Private Function HasNextNonEmptyRow(ByVal ws As Worksheet, ByVal currentRow As Long, ByVal lastRow As Long, ByVal lastCol As Long) As Boolean
    Dim r As Long

    For r = currentRow + 1 To lastRow
        If Not IsEntireRowEmpty(ws, r, lastCol) Then
            HasNextNonEmptyRow = True
            Exit Function
        End If
    Next r

    HasNextNonEmptyRow = False
End Function
