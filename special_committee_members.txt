Option Explicit

Public Sub ExportSpecialCommitteeMembersJs()

    Dim ws As Worksheet
    Dim lastRow As Long
    Dim r As Long

    Dim cCommitteeInstanceId As Long
    Dim cMeetingId As Long
    Dim cCommitteeId As Long
    Dim cMemberId As Long
    Dim cRole As Long
    Dim cStartDate As Long
    Dim cEndDate As Long

    Dim body As String
    Dim js As String
    Dim outputPath As String

    On Error GoTo EH

    Set ws = ThisWorkbook.Worksheets("special_committee_members")
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    If lastRow < 2 Then
        MsgBox "special_committee_members シートにデータがありません。", vbExclamation
        Exit Sub
    End If

    cCommitteeInstanceId = FindHeaderColumn(ws, "committee_instance_id")
    cMeetingId = FindHeaderColumn(ws, "meeting_id")
    cCommitteeId = FindHeaderColumn(ws, "committee_id")
    cMemberId = FindHeaderColumn(ws, "member_id")
    cRole = FindHeaderColumn(ws, "role")
    cStartDate = FindHeaderColumn(ws, "start_date")
    cEndDate = FindHeaderColumn(ws, "end_date")

    ValidateRequiredColumn cCommitteeInstanceId, "committee_instance_id"
    ValidateRequiredColumn cMeetingId, "meeting_id"
    ValidateRequiredColumn cCommitteeId, "committee_id"
    ValidateRequiredColumn cMemberId, "member_id"
    ValidateRequiredColumn cRole, "role"
    ValidateRequiredColumn cStartDate, "start_date"
    ValidateRequiredColumn cEndDate, "end_date"

    body = ""

    For r = 2 To lastRow
        If HasDataRow(ws, r, cCommitteeInstanceId, cMeetingId, cCommitteeId, cMemberId) Then
            If body <> "" Then body = body & "," & vbCrLf
            body = body & BuildObjectText( _
                ws, r, _
                cCommitteeInstanceId, _
                cMeetingId, _
                cCommitteeId, _
                cMemberId, _
                cRole, _
                cStartDate, _
                cEndDate _
            )
        End If
    Next r

    js = ""
    js = js & "window.APP_DATA = window.APP_DATA || {};" & vbCrLf
    js = js & "window.APP_DATA.special_committee_members = [" & vbCrLf
    js = js & body & vbCrLf
    js = js & "];" & vbCrLf

    EnsureFolder ThisWorkbook.Path & "\app"
    EnsureFolder ThisWorkbook.Path & "\app\data"
    EnsureFolder ThisWorkbook.Path & "\app\data\committees"

    outputPath = ThisWorkbook.Path & "\app\data\committees\special_committee_members.js"
    WriteTextFileUtf8 outputPath, js

    MsgBox "special_committee_members.js を出力しました。" & vbCrLf & outputPath, vbInformation
    Exit Sub

EH:
    MsgBox "エラー: " & Err.Description, vbCritical

End Sub

Private Function BuildObjectText( _
    ByVal ws As Worksheet, _
    ByVal rowIndex As Long, _
    ByVal cCommitteeInstanceId As Long, _
    ByVal cMeetingId As Long, _
    ByVal cCommitteeId As Long, _
    ByVal cMemberId As Long, _
    ByVal cRole As Long, _
    ByVal cStartDate As Long, _
    ByVal cEndDate As Long _
) As String

    Dim committeeInstanceId As String
    Dim meetingId As String
    Dim committeeId As String
    Dim memberId As String
    Dim roleName As String
    Dim startDate As String
    Dim endDate As String

    committeeInstanceId = CellText(ws.Cells(rowIndex, cCommitteeInstanceId).Value)
    meetingId = CellText(ws.Cells(rowIndex, cMeetingId).Value)
    committeeId = CellText(ws.Cells(rowIndex, cCommitteeId).Value)
    memberId = CellText(ws.Cells(rowIndex, cMemberId).Value)
    roleName = CellText(ws.Cells(rowIndex, cRole).Value)
    startDate = NormalizeDateText(ws.Cells(rowIndex, cStartDate).Value)
    endDate = NormalizeDateText(ws.Cells(rowIndex, cEndDate).Value)

    Dim s As String
    s = ""
    s = s & "  {" & vbCrLf
    s = s & "    committee_instance_id: """ & EscapeJs(committeeInstanceId) & """," & vbCrLf
    s = s & "    meeting_id: """ & EscapeJs(meetingId) & """," & vbCrLf
    s = s & "    committee_id: """ & EscapeJs(committeeId) & """," & vbCrLf
    s = s & "    member_id: """ & EscapeJs(memberId) & """," & vbCrLf
    s = s & "    role: """ & EscapeJs(roleName) & """," & vbCrLf
    s = s & "    start_date: """ & EscapeJs(startDate) & """," & vbCrLf
    s = s & "    end_date: """ & EscapeJs(endDate) & """" & vbCrLf
    s = s & "  }"

    BuildObjectText = s

End Function

Private Function HasDataRow( _
    ByVal ws As Worksheet, _
    ByVal rowIndex As Long, _
    ByVal cCommitteeInstanceId As Long, _
    ByVal cMeetingId As Long, _
    ByVal cCommitteeId As Long, _
    ByVal cMemberId As Long _
) As Boolean

    If CellText(ws.Cells(rowIndex, cCommitteeInstanceId).Value) = "" Then Exit Function
    If CellText(ws.Cells(rowIndex, cMeetingId).Value) = "" Then Exit Function
    If CellText(ws.Cells(rowIndex, cCommitteeId).Value) = "" Then Exit Function
    If CellText(ws.Cells(rowIndex, cMemberId).Value) = "" Then Exit Function

    HasDataRow = True

End Function

Private Function NormalizeDateText(ByVal v As Variant) As String

    If IsError(v) Then
        NormalizeDateText = ""
        Exit Function
    End If

    If IsDate(v) Then
        NormalizeDateText = Format$(CDate(v), "yyyy-mm-dd")
        Exit Function
    End If

    Dim s As String
    s = Trim$(CStr(v))

    If s = "" Then
        NormalizeDateText = ""
        Exit Function
    End If

    s = Replace(s, "/", "-")
    s = Replace(s, ".", "-")

    If IsDate(s) Then
        NormalizeDateText = Format$(CDate(s), "yyyy-mm-dd")
    Else
        NormalizeDateText = s
    End If

End Function

Private Function CellText(ByVal v As Variant) As String
    If IsError(v) Then
        CellText = ""
    Else
        CellText = Trim$(CStr(v))
    End If
End Function

Private Function EscapeJs(ByVal s As String) As String
    s = Replace(s, "\", "\\")
    s = Replace(s, """", "\""")
    s = Replace(s, vbCrLf, "\n")
    s = Replace(s, vbCr, "\n")
    s = Replace(s, vbLf, "\n")
    EscapeJs = s
End Function

Private Function FindHeaderColumn(ByVal ws As Worksheet, ByVal headerName As String) As Long

    Dim lastCol As Long
    Dim c As Long

    lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column

    For c = 1 To lastCol
        If Trim$(CStr(ws.Cells(1, c).Value)) = headerName Then
            FindHeaderColumn = c
            Exit Function
        End If
    Next c

    FindHeaderColumn = 0

End Function

Private Sub ValidateRequiredColumn(ByVal colIndex As Long, ByVal headerName As String)
    If colIndex = 0 Then
        Err.Raise vbObjectError + 1000, , "ヘッダが見つかりません: " & headerName
    End If
End Sub

Private Sub EnsureFolder(ByVal folderPath As String)
    If Dir(folderPath, vbDirectory) = "" Then MkDir folderPath
End Sub

Private Sub WriteTextFileUtf8(ByVal filePath As String, ByVal textData As String)
    Dim stm As Object
    Set stm = CreateObject("ADODB.Stream")

    stm.Type = 2
    stm.Charset = "UTF-8"
    stm.Open
    stm.WriteText textData
    stm.SaveToFile filePath, 2
    stm.Close

    Set stm = Nothing
End Sub
