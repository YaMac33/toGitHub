' -----------------------------
' 公開用エントリーポイント
' -----------------------------

Public Sub ExportAllData()
    Dim configs As Variant
    Dim i As Long

    On Error GoTo ErrHandler

    EnsureOutputFolderExists
    configs = GetAllSheetConfigs()

    For i = LBound(configs) To UBound(configs)
        ExportSheetData CStr(configs(i)(0))
    Next i

    MsgBox "全データの出力が完了しました。", vbInformation
    Exit Sub

ErrHandler:
    MsgBox "ExportAllData エラー: " & Err.Description, vbCritical
End Sub

Public Sub ExportActiveSheetData()
    On Error GoTo ErrHandler

    EnsureOutputFolderExists
    ExportSheetData ActiveSheet.Name

    MsgBox "アクティブシートの出力が完了しました。", vbInformation
    Exit Sub

ErrHandler:
    MsgBox "ExportActiveSheetData エラー: " & Err.Description, vbCritical
End Sub

Public Sub ExportMeetings()

End Sub

End Sub

Public Sub ExportSpecialCommitteeInstances()

Public Sub ExportSpecialCommitteeMembers()

' -----------------------------
' メイン処理
' -----------------------------

Public Sub ExportSheetData(ByVal sheetName As String)
    Dim ws As Worksheet
    Dim config As Variant
    Dim jsText As String
    Dim outputPath As String

    On Error GoTo ErrHandler

    Set ws = ThisWorkbook.Worksheets(sheetName)
    config = GetSheetConfig(sheetName)

    jsText = BuildJsFromSheet(ws, CStr(config(2)))
    outputPath = GetOutputFolderPath() & "\" & CStr(config(1))

    WriteTextFile outputPath, jsText

    Debug.Print "Exported: " & sheetName & " -> " & outputPath
    Exit Sub

ErrHandler:
    MsgBox "ExportSheetData エラー [" & sheetName & "]: " & Err.Description, vbCritical
End Sub

' -----------------------------
' シート → JS文字列変換
' -----------------------------

Private Function BuildJsFromSheet(ByVal ws As Worksheet, ByVal dataKey As String) As String
    Dim lastRow As Long
    Dim lastCol As Long
    Dim r As Long
    Dim c As Long
    Dim sb As String
    Dim headers() As String
    Dim rowText As String
    Dim cellValue As Variant
    Dim headerName As String

    lastRow = GetLastDataRow(ws)
    lastCol = GetLastDataCol(ws)

    If lastCol = 0 Then
        Err.Raise vbObjectError + 1001, , "ヘッダ行が見つかりません。"
    End If

    ReDim headers(1 To lastCol)

    For c = 1 To lastCol
        headers(c) = Trim$(CStr(ws.Cells(1, c).Value))
        If headers(c) = "" Then
            Err.Raise vbObjectError + 1002, , "1行目に空のヘッダがあります。列=" & c
        End If
    Next c

    sb = ""
    sb = sb & "window.APP_DATA = window.APP_DATA || {};" & vbCrLf
    sb = sb & "window.APP_DATA." & dataKey & " = [" & vbCrLf

    If lastRow >= 2 Then
        For r = 2 To lastRow
            If Not IsEntireRowEmpty(ws, r, lastCol) Then
                rowText = "  {" & vbCrLf

                For c = 1 To lastCol
                    headerName = headers(c)
                    cellValue = ws.Cells(r, c).Value

                    rowText = rowText & "    " & headerName & ": " & ConvertCellToJsValue(cellValue, headerName)

                    If c < lastCol Then
                        rowText = rowText & ","
                    End If

                    rowText = rowText & vbCrLf
                Next c

                rowText = rowText & "  }"

                If HasNextNonEmptyRow(ws, r, lastRow, lastCol) Then
                    rowText = rowText & ","
                End If

                rowText = rowText & vbCrLf
                sb = sb & rowText
            End If
        Next r
    End If

    sb = sb & "];" & vbCrLf

    BuildJsFromSheet = sb
End Function

' -----------------------------
' 値変換
' -----------------------------

Private Function ConvertCellToJsValue(ByVal v As Variant, ByVal headerName As String) As String
    If IsError(v) Then
        ConvertCellToJsValue = """"""
        Exit Function
    End If

    If IsEmpty(v) Then
        ConvertCellToJsValue = """"""
        Exit Function
    End If

    If Trim$(CStr(v)) = "" Then
        ConvertCellToJsValue = """"""
        Exit Function
    End If

    If IsBooleanField(headerName) Then
        ConvertCellToJsValue = NormalizeBooleanLike(v)
        Exit Function
    End If

    If IsNumericField(headerName) Then
        ConvertCellToJsValue = NormalizeNumericLike(v)
        Exit Function
    End If

    If IsDateField(headerName) Then
        ConvertCellToJsValue = """" & EscapeJsString(FormatDateForJs(v, headerName)) & """"
        Exit Function
    End If

    ConvertCellToJsValue = """" & EscapeJsString(CStr(v)) & """"
End Function

Private Function IsBooleanField(ByVal headerName As String) As Boolean
    Select Case headerName
        Case "active_flag"
            IsBooleanField = True
        Case Else
            IsBooleanField = False
    End Select
End Function

Private Function IsNumericField(ByVal headerName As String) As Boolean
    Select Case headerName
        Case "fiscal_year", "year", "term_no", "sort_order"
            IsNumericField = True
        Case Else
            IsNumericField = False
    End Select
End Function

Private Function IsDateField(ByVal headerName As String) As Boolean
    Dim lowerName As String
    lowerName = LCase$(headerName)

    If Right$(lowerName, 5) = "_date" Then
        IsDateField = True
        Exit Function
    End If

    If Right$(lowerName, 3) = "_at" Then
        IsDateField = True
        Exit Function
    End If

    IsDateField = False
End Function

Private Function NormalizeBooleanLike(ByVal v As Variant) As String
    Dim s As String
    s = LCase$(Trim$(CStr(v)))

    Select Case s
        Case "1", "true", "yes", "y"
            NormalizeBooleanLike = "1"
        Case "0", "false", "no", "n"
            NormalizeBooleanLike = "0"
        Case Else
            If IsNumeric(v) Then
                If CLng(v) <> 0 Then
                    NormalizeBooleanLike = "1"
                Else
                    NormalizeBooleanLike = "0"
                End If
            Else
                NormalizeBooleanLike = "0"
            End If
    End Select
End Function

Private Function NormalizeNumericLike(ByVal v As Variant) As String
    If IsNumeric(v) Then
        If CDbl(v) = Int(CDbl(v)) Then
            NormalizeNumericLike = CStr(CLng(v))
        Else
            NormalizeNumericLike = Replace(CStr(CDbl(v)), ",", "")
        End If
    Else
        NormalizeNumericLike = "0"
    End If
End Function

Private Function FormatDateForJs(ByVal v As Variant, ByVal headerName As String) As String
    If IsDate(v) Then
        If LCase$(headerName) Like "*_at" Then
            FormatDateForJs = Format$(CDate(v), "yyyy-mm-dd hh:nn:ss")
        Else
            FormatDateForJs = Format$(CDate(v), "yyyy-mm-dd")
        End If
    Else
        FormatDateForJs = Trim$(CStr(v))
    End If
End Function

Private Function EscapeJsString(ByVal s As String) As String
    Dim t As String
    t = s
    t = Replace(t, "\", "\\")
    t = Replace(t, """", "\""")
    t = Replace(t, vbCrLf, "\n")
    t = Replace(t, vbCr, "\n")
    t = Replace(t, vbLf, "\n")
    EscapeJsString = t
End Function

' -----------------------------
' シート設定
' -----------------------------

Private Function GetAllSheetConfigs() As Variant
    GetAllSheetConfigs = Array( _
        Array("会議回マスタ", "meetings.js", "meetings"), _
        Array("イベント種別マスタ", "event_types.js", "event_types"), _
        Array("議会イベント履歴", "events.js", "events"), _
        Array("特別委員会マスタ", "special_committees.js", "special_committees"), _
        Array("特別委員会設置履歴", "special_committee_instances.js", "special_committee_instances"), _
        Array("特別委員会会議日", "special_committee_meetings.js", "special_committee_meetings"), _
        Array("特別委員会委員名簿履歴", "special_committee_members.js", "special_committee_members") _
    )
End Function

Private Function GetSheetConfig(ByVal sheetName As String) As Variant
    Dim configs As Variant
    Dim i As Long

    configs = GetAllSheetConfigs()

    For i = LBound(configs) To UBound(configs)
        If CStr(configs(i)(0)) = sheetName Then
            GetSheetConfig = configs(i)
            Exit Function
        End If
    Next i

    Err.Raise vbObjectError + 1101, , "シート設定が見つかりません: " & sheetName
End Function

' -----------------------------
' パス・ファイル出力
' -----------------------------

Private Function GetOutputFolderPath() As String
    GetOutputFolderPath = ThisWorkbook.Path & "\app\data"
End Function

Private Sub EnsureOutputFolderExists()
    Dim fso As Object
    Dim folderPath As String
    Dim appPath As String

    folderPath = GetOutputFolderPath()
    appPath = ThisWorkbook.Path & "\app"

    Set fso = CreateObject("Scripting.FileSystemObject")

    If Not fso.FolderExists(appPath) Then
        Err.Raise vbObjectError + 1201, , "app フォルダがありません: " & appPath
    End If

    If Not fso.FolderExists(folderPath) Then
        fso.CreateFolder folderPath
    End If
End Sub

Private Sub WriteTextFile(ByVal filePath As String, ByVal textContent As String)
    Dim stm As Object

    Set stm = CreateObject("ADODB.Stream")
    stm.Type = 2
    stm.Charset = "utf-8"
    stm.Open
    stm.WriteText textContent
    stm.SaveToFile filePath, 2
    stm.Close
End Sub

' -----------------------------
' シート走査補助
' -----------------------------

Private Function GetLastDataRow(ByVal ws As Worksheet) As Long
    Dim lastCell As Range

    On Error Resume Next
    Set lastCell = ws.Cells.Find(What:="*", After:=ws.Cells(1, 1), LookIn:=xlFormulas, _
                                 LookAt:=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlPrevious)
    On Error GoTo 0

    If lastCell Is Nothing Then
        GetLastDataRow = 0
    Else
        GetLastDataRow = lastCell.Row
    End If
End Function

Private Function GetLastDataCol(ByVal ws As Worksheet) As Long
    Dim lastCell As Range

    On Error Resume Next
    Set lastCell = ws.Rows(1).Find(What:="*", After:=ws.Cells(1, 1), LookIn:=xlFormulas, _
                                   LookAt:=xlPart, SearchOrder:=xlByColumns, SearchDirection:=xlPrevious)
    On Error GoTo 0

    If lastCell Is Nothing Then
        GetLastDataCol = 0
    Else
        GetLastDataCol = lastCell.Column
    End If
End Function

Private Function IsEntireRowEmpty(ByVal ws As Worksheet, ByVal rowNum As Long, ByVal lastCol As Long) As Boolean
    Dim c As Long

    For c = 1 To lastCol
        If Trim$(CStr(ws.Cells(rowNum, c).Value)) <> "" Then
            IsEntireRowEmpty = False
            Exit Function
        End If
    Next c

    IsEntireRowEmpty = True
End Function

Private Function HasNextNonEmptyRow(ByVal ws As Worksheet, ByVal currentRow As Long, ByVal lastRow As Long, ByVal lastCol As Long) As Boolean
    Dim r As Long

    For r = currentRow + 1 To lastRow
        If Not IsEntireRowEmpty(ws, r, lastCol) Then
            HasNextNonEmptyRow = True
            Exit Function
        End If
    Next r

    HasNextNonEmptyRow = False
End Function
