Private Sub AddHeader(sld As Slide, eyebrow As String, title As String, badge As String)

'==================================================================
' 汎用テキストボックス
'==================================================================
Private Function MakeText(sld As Slide, x As Single, y As Single, w As Single, h As Single, _
    txt As String, sz As Single, bold As Boolean, colorC As Long, _
    Optional align As Long = msoAlignLeft, Optional anchor As Long = msoAnchorTop) As Shape

    Dim shp As Shape
    Set shp = sld.Shapes.AddTextbox(msoTextOrientationHorizontal, x, y, w, h)
    With shp.TextFrame2
        .WordWrap = msoTrue
        .AutoSize = msoAutoSizeNone
        .MarginLeft = 0: .MarginRight = 0: .MarginTop = 0: .MarginBottom = 0
        .VerticalAnchor = anchor
        With .TextRange
            .Text = txt
            .Font.Size = sz
            .Font.Bold = IIf(bold, msoTrue, msoFalse)
            .Font.Name = mJpFont
            .Font.Fill.ForeColor.RGB = colorC
            .ParagraphFormat.Alignment = align
            .ParagraphFormat.SpaceWithin = 1   ' 行間（行送り）を読みやすく
        End With
    End With
    Set MakeText = shp
End Function

'==================================================================
' 汎用角丸長方形
'==================================================================
Private Function MakeRect(sld As Slide, x As Single, y As Single, w As Single, h As Single, _
    fillC As Long, lineC As Long, Optional radius As Single = 0.08) As Shape

    Dim shp As Shape
    Set shp = sld.Shapes.AddShape(msoShapeRoundedRectangle, x, y, w, h)
    On Error Resume Next
    shp.Adjustments(1) = radius
    On Error GoTo 0
    shp.Fill.ForeColor.RGB = fillC
    shp.Line.ForeColor.RGB = lineC
    shp.Line.Weight = 1
    shp.Shadow.Visible = msoFalse
    Set MakeRect = shp
End Function

'==================================================================
' バリアント色の取得（カード種別→背景色・枠色・見出し色）
'==================================================================
Private Sub GetVariant(variant As String, ByRef fillC As Long, ByRef lineC As Long, ByRef headC As Long)
    Select Case LCase(variant)
        Case "primary"
            fillC = mPrimarySoft: lineC = mPrimary: headC = mPrimaryDark
        Case "accent"
            fillC = mAccentSoft: lineC = mAccent: headC = RGB(138, 90, 10)
        Case "green"
            fillC = mGreenSoft: lineC = mGreen: headC = mGreen
        Case "red"
            fillC = mRedSoft: lineC = mRed: headC = RGB(117, 48, 48)
        Case Else   ' white
            fillC = mPaper: lineC = mLine: headC = mPrimaryDark
    End Select
End Sub

'==================================================================
' カード（見出し＋本文。numを指定すると番号付き円を表示）
'==================================================================
Private Function AddCard(sld As Slide, x As Single, y As Single, w As Single, h As Single, _
    heading As String, body As String, variant As String, Optional num As Integer = 0) As Shape

    Dim fillC As Long, lineC As Long, headC As Long
    GetVariant variant, fillC, lineC, headC

    Dim r As Shape
    Set r = MakeRect(sld, x, y, w, h, fillC, lineC, 0.06)

    Dim ty As Single
    ty = y + 14

    ' 番号付き円
    If num > 0 Then
        Dim ov As Shape
        Set ov = sld.Shapes.AddShape(msoShapeOval, x + 16, y + 14, 28, 28)
        ov.Fill.ForeColor.RGB = mPrimary
        ov.Line.Visible = msoFalse
        ov.Shadow.Visible = msoFalse
        With ov.TextFrame2
            .VerticalAnchor = msoAnchorMiddle
            .MarginLeft = 0: .MarginRight = 0: .MarginTop = 0: .MarginBottom = 0
            With .TextRange
                .Text = CStr(num)
                .Font.Size = 14
                .Font.Bold = msoTrue
                .Font.Name = mJpFont
                .Font.Fill.ForeColor.RGB = mPaper
                .ParagraphFormat.Alignment = msoAlignCenter
            End With
        End With
        ty = y + 50
    End If

    ' 見出し
    MakeText sld, x + 16, ty, w - 32, 24, heading, 14.5, True, headC, msoAlignLeft, msoAnchorTop
    ' 本文
    MakeText sld, x + 16, ty + 27, w - 32, h - (ty + 27 - y) - 14, body, 12, False, mInk, msoAlignLeft, msoAnchorTop

    Set AddCard = r
End Function

'==================================================================
' フロー（横並びのステップ。間に「›」を表示）
'==================================================================
Private Sub AddFlow(sld As Slide, x As Single, y As Single, totalW As Single, h As Single, _
    titles() As String, descs() As String)

    Dim n As Integer
    n = UBound(titles) - LBound(titles) + 1
    Dim gap As Single
    gap = 22
    Dim sw As Single
    sw = (totalW - gap * (n - 1)) / n

    Dim i As Integer
    For i = 0 To n - 1
        Dim sx As Single
        sx = x + i * (sw + gap)
        MakeRect sld, sx, y, sw, h, mPaper, mLine, 0.07

        ' 番号付き円
        Dim ov As Shape
        Set ov = sld.Shapes.AddShape(msoShapeOval, sx + 16, y + 14, 28, 28)
        ov.Fill.ForeColor.RGB = mPrimary
        ov.Line.Visible = msoFalse
        ov.Shadow.Visible = msoFalse
        With ov.TextFrame2
            .VerticalAnchor = msoAnchorMiddle
            With .TextRange
                .Text = CStr(i + 1)
                .Font.Size = 14: .Font.Bold = msoTrue
                .Font.Name = mJpFont
                .Font.Fill.ForeColor.RGB = mPaper
                .ParagraphFormat.Alignment = msoAlignCenter
            End With
        End With

        ' ステップ名
        MakeText sld, sx + 14, y + 50, sw - 28, 24, titles(i), 13.5, True, mPrimaryDark, msoAlignLeft, msoAnchorTop
        ' 説明
        MakeText sld, sx + 14, y + 76, sw - 28, h - 86, descs(i), 11, False, mInk, msoAlignLeft, msoAnchorTop

        ' ステップ間の矢印
        If i < n - 1 Then
            MakeText sld, sx + sw, y, gap, h, ">", 24, True, mPrimary, msoAlignCenter, msoAnchorMiddle
        End If
    Next i
End Sub

'==================================================================
' 目次（2列）
'==================================================================
Private Sub AddTOC(sld As Slide, x As Single, y As Single, w As Single, h As Single, items() As String)
    Dim n As Integer
    n = UBound(items) - LBound(items) + 1
    Dim cols As Integer
    cols = 2
    Dim rows As Integer
    rows = (n + cols - 1) \ cols

    Dim gx As Single, gy As Single
    gx = 24: gy = 14
    Dim cw As Single, ch As Single
    cw = (w - gx * (cols - 1)) / cols
    ch = (h - gy * (rows - 1)) / rows

    Dim i As Integer
    For i = 0 To n - 1
        Dim c As Integer, rr As Integer
        c = i Mod cols
        rr = i \ cols
        Dim ix As Single, iy As Single
        ix = x + c * (cw + gx)
        iy = y + rr * (ch + gy)

        MakeRect sld, ix, iy, cw, ch, mPaper, mLine, 0.12

        ' 番号円
        Dim ov As Shape
        Set ov = sld.Shapes.AddShape(msoShapeOval, ix + 14, iy + (ch - 30) / 2, 30, 30)
        ov.Fill.ForeColor.RGB = mPrimary
        ov.Line.Visible = msoFalse
        ov.Shadow.Visible = msoFalse
        With ov.TextFrame2
            .VerticalAnchor = msoAnchorMiddle
            With .TextRange
                .Text = CStr(i + 1)
                .Font.Size = 13: .Font.Bold = msoTrue
                .Font.Name = mJpFont
                .Font.Fill.ForeColor.RGB = mPaper
                .ParagraphFormat.Alignment = msoAlignCenter
            End With
        End With

        ' 項目名
        MakeText sld, ix + 56, iy, cw - 70, ch, items(i), 14, True, mInk, msoAlignLeft, msoAnchorMiddle
    Next i
End Sub

'==================================================================
' 比較表（ヘッダー1行＋本文）
'==================================================================
Private Sub AddComparisonTable(sld As Slide, x As Single, y As Single, w As Single, h As Single, _
    hdr() As String, body() As String)

    Dim cols As Integer, rows As Integer
    cols = UBound(hdr) - LBound(hdr) + 1
    rows = UBound(body, 1) - LBound(body, 1) + 2   ' データ行数＋ヘッダー

    Dim shpT As Shape
    Set shpT = sld.Shapes.AddTable(rows, cols, x, y, w, h)
    Dim tbl As Table
    Set tbl = shpT.Table

    ' 列幅（1列目を狭め、2・3列目を広く）
    tbl.Columns(1).Width = w * 0.2
    tbl.Columns(2).Width = w * 0.42
    tbl.Columns(3).Width = w * 0.38

    Dim c As Integer, rr As Integer
    ' ヘッダー行
    For c = 1 To cols
        With tbl.Cell(1, c)
            .Shape.Fill.ForeColor.RGB = mPrimarySoft
            With .Shape.TextFrame2.TextRange
                .Text = hdr(c - 1)
                .Font.Size = 13
                .Font.Bold = msoTrue
                .Font.Name = mJpFont
                .Font.Fill.ForeColor.RGB = mPrimaryDark
            End With
            .Shape.TextFrame2.VerticalAnchor = msoAnchorMiddle
        End With
    Next c

    ' データ行
    For rr = 1 To rows - 1
        For c = 1 To cols
            With tbl.Cell(rr + 1, c)
                .Shape.Fill.ForeColor.RGB = mPaper
                With .Shape.TextFrame2.TextRange
                    .Text = body(rr - 1, c - 1)
                    .Font.Size = 12
                    .Font.Bold = IIf(c = 1, msoTrue, msoFalse)
                    .Font.Name = mJpFont
                    .Font.Fill.ForeColor.RGB = IIf(c = 1, mPrimaryDark, mInk)
                End With
                .Shape.TextFrame2.VerticalAnchor = msoAnchorMiddle
            End With
        Next c
    Next rr
End Sub

' アイコン風リスト項目（左に「!」マーク＋見出し＋説明）
'==================================================================
Private Sub AddIconItem(sld As Slide, x As Single, y As Single, w As Single, h As Single, _
    heading As String, desc As String)

    MakeRect sld, x, y, w, h, mPaper, mLine, 0.1

    ' アイコン枠
    Dim icn As Shape
    Set icn = MakeRect(sld, x + 16, y + (h - 40) / 2, 40, 40, mPrimarySoft, mPrimarySoft, 0.2)
    With icn.TextFrame2
        .VerticalAnchor = msoAnchorMiddle
        With .TextRange
            .Text = "!"
            .Font.Size = 20: .Font.Bold = msoTrue
            .Font.Name = mJpFont
            .Font.Fill.ForeColor.RGB = mPrimaryDark
            .ParagraphFormat.Alignment = msoAlignCenter
        End With
    End With

    ' 見出し
    MakeText sld, x + 72, y + 14, w - 90, 24, heading, 14.5, True, mPrimaryDark, msoAlignLeft, msoAnchorTop
    ' 説明
    MakeText sld, x + 72, y + 40, w - 90, h - 50, desc, 12, False, mInk, msoAlignLeft, msoAnchorTop
End Sub

'==================================================================
' ノート欄に読み原稿を設定
'==================================================================
Private Sub AddNotes(sld As Slide, txt As String)
    Dim shp As Shape
    On Error Resume Next
    For Each shp In sld.NotesPage.Shapes
        If shp.PlaceholderFormat.Type = ppPlaceholderBody Then
            shp.TextFrame.TextRange.Text = txt
            Exit For
        End If
    Next shp
    On Error GoTo 0
End Sub
