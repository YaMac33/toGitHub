Option Explicit

Public Sub CreateEmptyFilesFromList()

    Dim ws As Worksheet
    Dim lastRow As Long
    Dim i As Long
    Dim fileName As String
    Dim fullPath As String

    On Error GoTo ErrHandler

    Set ws = ActiveSheet
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    For i = 1 To lastRow
        fileName = Trim(CStr(ws.Cells(i, 1).Value))

        If fileName <> "" Then
            fullPath = ThisWorkbook.Path & "\" & Replace(fileName, "/", "\")
            CreateFoldersRecursive fullPath
            CreateEmptyFile fullPath
        End If
    Next i

    MsgBox "空ファイルの作成が完了しました。", vbInformation
    Exit Sub

ErrHandler:
    MsgBox "エラー: " & Err.Description, vbCritical
End Sub

Private Sub CreateFoldersRecursive(ByVal filePath As String)

    Dim fso As Object
    Dim folderPath As String
    Dim parts() As String
    Dim i As Long
    Dim currentPath As String

    Set fso = CreateObject("Scripting.FileSystemObject")

    folderPath = Left(filePath, InStrRev(filePath, "\") - 1)
    parts = Split(folderPath, "\")

    currentPath = parts(0)

    For i = 1 To UBound(parts)
        currentPath = currentPath & "\" & parts(i)
        If Not fso.FolderExists(currentPath) Then
            fso.CreateFolder currentPath
        End If
    Next i

End Sub

Private Sub CreateEmptyFile(ByVal filePath As String)

    Dim fso As Object
    Dim fileObj As Object

    Set fso = CreateObject("Scripting.FileSystemObject")
    Set fileObj = fso.CreateTextFile(filePath, True)

    fileObj.Write ""
    fileObj.Close

End Sub
