Option Explicit

Public Sub CreateEmptyFilesFromList()

    Dim ws As Worksheet
    Dim lastRow As Long
    Dim i As Long
    Dim fileName As String
    Dim fullPath As String

    Set ws = ActiveSheet
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    For i = 1 To lastRow

        fileName = Trim(ws.Cells(i, 1).Value)

        If fileName <> "" Then

            fullPath = ThisWorkbook.Path & "\" & fileName

            ' フォルダが無ければ作成
            Call CreateFolderIfNeeded(fullPath)

            ' 空ファイル作成
            Call CreateEmptyFile(fullPath)

        End If

    Next i

    MsgBox "空ファイルの作成が完了しました。", vbInformation

End Sub
