Option Explicit

' =========================================================
' カレンダー用
' Excelシート → calendar/events.js 出力
'
' 前提
' - 1行目: 列の説明
' - 2行目: 列名
' - 3行目以降: データ
' - このブックと同じ階層に calendar フォルダがある
'
' 想定列
' event_id
' date
' end_date
' title
' start_time
' end_time
' is_all_day
' category
' department_name
' location
' note
' sort_order
' =========================================================

Public Sub ExportCalendarEvents()
    On Error GoTo ErrHandler

    Dim ws As Worksheet
    Dim lastRow As Long
    Dim lastCol As Long
    Dim r As Long
    Dim c As Long

    Dim basePath As String
    Dim outPath As String

    Dim fso As Object
    Dim stream As Object

    Dim headerMap As Object
    Dim lineText As String
    Dim isFirst As Boolean

    ' ▼必要に応じてシート名を変更
    Set ws = ThisWorkbook.Worksheets("calendar")

    lastRow = GetLastRow(ws)
    lastCol = GetLastCol(ws)

    If lastRow < 3 Then
        MsgBox "データ行がありません。", vbExclamation
        Exit Sub
    End If

    Set headerMap = CreateObject("Scripting.Dictionary")
    BuildHeaderMap ws, headerMap, lastCol

    ' 必須列チェック
    CheckRequiredHeader headerMap, "date"
    CheckRequiredHeader headerMap, "title"

    basePath = ThisWorkbook.Path
    outPath = basePath & "\calendar\events.js"

    Set fso = CreateObject("Scripting.FileSystemObject")

    If Not fso.FolderExists(basePath & "\calendar") Then
        MsgBox "出力先フォルダが存在しません: " & vbCrLf & basePath & "\calendar", vbExclamation
        Exit Sub
    End If

    Set stream = CreateObject("ADODB.Stream")
    stream.Type = 2                  ' text
    stream.Charset = "utf-8"
    stream.Open

    stream.WriteText "window.APP_DATA = window.APP_DATA || {};" & vbCrLf & vbCrLf
    stream.WriteText "window.APP_DATA.events = [" & vbCrLf & vbCrLf

    isFirst = True

    For r = 3 To lastRow
        If IsSkippableRow(ws, r, headerMap) Then
            GoTo ContinueLoop
        End If

        If Not isFirst Then
            stream.WriteText "," & vbCrLf
        End If

        lineText = BuildEventObjectText(ws, r, headerMap)
        stream.WriteText lineText

        isFirst = False

ContinueLoop:
    Next r

    stream.WriteText vbCrLf & vbCrLf & "];" & vbCrLf

    If fso.FileExists(outPath) Then
        fso.DeleteFile outPath, True
    End If

    stream.SaveToFile outPath, 2   ' adSaveCreateOverWrite
    stream.Close

    MsgBox "出力完了:" & vbCrLf & outPath, vbInformation
    Exit Sub

ErrHandler:
    On Error Resume Next
    If Not stream Is Nothing Then
        If stream.State = 1 Then stream.Close
    End If
    MsgBox "エラー: " & Err.Description, vbCritical
End Sub

Private Function BuildEventObjectText(ByVal ws As Worksheet, ByVal rowIndex As Long, ByVal headerMap As Object) As String
    Dim eventId As String
    Dim eventDate As String
    Dim endDate As String
    Dim title As String
    Dim startTime As String
    Dim endTime As String
    Dim isAllDay As Long
    Dim category As String
    Dim departmentName As String
    Dim location As String
    Dim note As String
    Dim sortOrder As Long

    eventId = GetCellByHeader(ws, rowIndex, headerMap, "event_id")
    eventDate = NormalizeDateValue(GetCellByHeader(ws, rowIndex, headerMap, "date"))
    endDate = NormalizeDateValue(GetCellByHeader(ws, rowIndex, headerMap, "end_date"))
    title = GetCellByHeader(ws, rowIndex, headerMap, "title")
    startTime = NormalizeTimeValue(GetCellByHeader(ws, rowIndex, headerMap, "start_time"))
    endTime = NormalizeTimeValue(GetCellByHeader(ws, rowIndex, headerMap, "end_time"))
    isAllDay = NormalizeFlagValue(GetCellByHeader(ws, rowIndex, headerMap, "is_all_day"))
    category = GetCellByHeader(ws, rowIndex, headerMap, "category")
    departmentName = GetCellByHeader(ws, rowIndex, headerMap, "department_name")
    location = GetCellByHeader(ws, rowIndex, headerMap, "location")
    note = GetCellByHeader(ws, rowIndex, headerMap, "note")
    sortOrder = NormalizeLongValue(GetCellByHeader(ws, rowIndex, headerMap, "sort_order"), 0)

    If Trim$(eventId) = "" Then
        eventId = BuildAutoEventId(eventDate, rowIndex)
    End If

    If Trim$(category) = "" Then
        category = "OTHER"
    End If

    BuildEventObjectText = _
        "  {" & vbCrLf & _
        "    event_id: """ & EscapeJs(eventId) & """," & vbCrLf & _
        "    date: """ & EscapeJs(eventDate) & """," & vbCrLf & _
        "    end_date: """ & EscapeJs(endDate) & """," & vbCrLf & _
        "    title: """ & EscapeJs(title) & """," & vbCrLf & _
        "    start_time: """ & EscapeJs(startTime) & """," & vbCrLf & _
        "    end_time: """ & EscapeJs(endTime) & """," & vbCrLf & _
        "    is_all_day: " & CStr(isAllDay) & "," & vbCrLf & _
        "    category: """ & EscapeJs(category) & """," & vbCrLf & _
        "    department_name: """ & EscapeJs(departmentName) & """," & vbCrLf & _
        "    location: """ & EscapeJs(location) & """," & vbCrLf & _
        "    note: """ & EscapeJs(note) & """," & vbCrLf & _
        "    sort_order: " & CStr(sortOrder) & vbCrLf & _
        "  }"
End Function

Private Function IsSkippableRow(ByVal ws As Worksheet, ByVal rowIndex As Long, ByVal headerMap As Object) As Boolean
    Dim vDate As String
    Dim vTitle As String

    vDate = Trim$(GetCellByHeader(ws, rowIndex, headerMap, "date"))
    vTitle = Trim$(GetCellByHeader(ws, rowIndex, headerMap, "title"))

    ' date と title が両方空なら空行扱い
    If vDate = "" And vTitle = "" Then
        IsSkippableRow = True
        Exit Function
    End If

    ' title が空ならスキップ
    If vTitle = "" Then
        IsSkippableRow = True
        Exit Function
    End If

    ' date が空ならスキップ
    If vDate = "" Then
        IsSkippableRow = True
        Exit Function
    End If

    IsSkippableRow = False
End Function

Private Sub BuildHeaderMap(ByVal ws As Worksheet, ByVal headerMap As Object, ByVal lastCol As Long)
    Dim c As Long
    Dim key As String

    For c = 1 To lastCol
        key = LCase$(Trim$(CStr(ws.Cells(2, c).Value)))
        If key <> "" Then
            If Not headerMap.Exists(key) Then
                headerMap.Add key, c
            End If
        End If
    Next c
End Sub

Private Sub CheckRequiredHeader(ByVal headerMap As Object, ByVal headerName As String)
    If Not headerMap.Exists(LCase$(headerName)) Then
        Err.Raise vbObjectError + 1000, , "必須列が見つかりません: " & headerName
    End If
End Sub

Private Function GetCellByHeader(ByVal ws As Worksheet, ByVal rowIndex As Long, ByVal headerMap As Object, ByVal headerName As String) As String
    Dim key As String
    key = LCase$(Trim$(headerName))

    If headerMap.Exists(key) Then
        GetCellByHeader = Trim$(CStr(ws.Cells(rowIndex, headerMap(key)).Value))
    Else
        GetCellByHeader = ""
    End If
End Function

Private Function NormalizeDateValue(ByVal valueText As String) As String
    If Trim$(valueText) = "" Then
        NormalizeDateValue = ""
        Exit Function
    End If

    If IsDate(valueText) Then
        NormalizeDateValue = Format$(CDate(valueText), "yyyy-mm-dd")
    Else
        NormalizeDateValue = Trim$(valueText)
    End If
End Function

Private Function NormalizeTimeValue(ByVal valueText As String) As String
    If Trim$(valueText) = "" Then
        NormalizeTimeValue = ""
        Exit Function
    End If

    If IsDate(valueText) Then
        NormalizeTimeValue = Format$(CDate(valueText), "hh:nn")
    Else
        NormalizeTimeValue = Trim$(valueText)
    End If
End Function

Private Function NormalizeFlagValue(ByVal valueText As String) As Long
    Dim t As String
    t = UCase$(Trim$(valueText))

    Select Case t
        Case "1", "TRUE", "終日", "YES", "Y"
            NormalizeFlagValue = 1
        Case Else
            NormalizeFlagValue = 0
    End Select
End Function

Private Function NormalizeLongValue(ByVal valueText As String, Optional ByVal defaultValue As Long = 0) As Long
    If Trim$(valueText) = "" Then
        NormalizeLongValue = defaultValue
    ElseIf IsNumeric(valueText) Then
        NormalizeLongValue = CLng(valueText)
    Else
        NormalizeLongValue = defaultValue
    End If
End Function

Private Function BuildAutoEventId(ByVal eventDate As String, ByVal rowIndex As Long) As String
    Dim ymd As String

    ymd = Replace$(eventDate, "-", "")
    If ymd = "" Then
        ymd = "00000000"
    End If

    BuildAutoEventId = "EVT" & ymd & "_" & Format$(rowIndex - 2, "00")
End Function

Private Function EscapeJs(ByVal s As String) As String
    Dim t As String
    t = s

    t = Replace$(t, "\", "\\")
    t = Replace$(t, """", "\""")
    t = Replace$(t, vbCrLf, "\n")
    t = Replace$(t, vbCr, "\n")
    t = Replace$(t, vbLf, "\n")

    EscapeJs = t
End Function

Private Function GetLastRow(ByVal ws As Worksheet) As Long
    Dim lastCell As Range
    Set lastCell = ws.Cells.Find(What:="*", After:=ws.Cells(1, 1), LookIn:=xlFormulas, _
                                 LookAt:=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlPrevious)
    If lastCell Is Nothing Then
        GetLastRow = 1
    Else
        GetLastRow = lastCell.Row
    End If
End Function

Private Function GetLastCol(ByVal ws As Worksheet) As Long
    Dim lastCell As Range
    Set lastCell = ws.Cells.Find(What:="*", After:=ws.Cells(1, 1), LookIn:=xlFormulas, _
                                 LookAt:=xlPart, SearchOrder:=xlByColumns, SearchDirection:=xlPrevious)
    If lastCell Is Nothing Then
        GetLastCol = 1
    Else
        GetLastCol = lastCell.Column
    End If
End Function
