Private Sub AddTitleSlide(ByVal pres As Presentation, ByVal titleText As String, ByVal subTitle As String, ByVal noteText As String, ByVal speakerNote As String)
    Dim sld As Slide
    Set sld = pres.Slides.Add(pres.Slides.Count + 1, ppLayoutBlank)
    ApplyBaseDesign sld

    AddTextBox sld, "Information Sharing / DX Trial Report", 70, 110, 760, 24, 13, RGB(31, 95, 139), True
    AddTextBox sld, titleText, 70, 145, 800, 120, 35, RGB(23, 63, 95), True
    AddTextBox sld, subTitle, 70, 275, 770, 80, 22, RGB(51, 65, 85), False

    AddAccentNote sld, noteText, 70, 380, 720, 58
    AddSpeakerNotes sld, speakerNote
End Sub

Private Sub AddAgendaSlide(ByVal pres As Presentation)
    Dim sld As Slide
    Set sld = pres.Slides.Add(pres.Slides.Count + 1, ppLayoutBlank)
    ApplyBaseDesign sld
    AddHeader sld, "Agenda", "本資料の流れ", "全体像"

    Dim items As Variant
    items = Array("背景・目的", "実験から分かったこと", "HTML・JavaScript活用", "資料作成への応用", "AIエージェントとGitHub", "GASによる自動化", "注意点・リスク", "まとめ・次のアクション")

    Dim i As Long, x As Single, y As Single
    For i = 0 To 7
        x = 80 + (i Mod 2) * 400
        y = 145 + Int(i / 2) * 72
        AddAgendaItem sld, CStr(i + 1), CStr(items(i)), x, y
    Next i

    AddSpeakerNotes sld, "本資料では、背景と目的から始め、今回の実験で分かったこと、HTMLやJavaScriptの活用、資料作成への応用、AIエージェントやGitHub、GASによる自動化、注意点とリスク、最後にまとめと次のアクションを整理します。"
End Sub

Private Sub AddTwoCardSlide(ByVal pres As Presentation, ByVal titleText As String, ByVal badgeText As String, _
                            ByVal card1Title As String, ByVal card1Body As String, _
                            ByVal card2Title As String, ByVal card2Body As String, _
                            ByVal noteBoxText As String, ByVal speakerNote As String)
    Dim sld As Slide
    Set sld = pres.Slides.Add(pres.Slides.Count + 1, ppLayoutBlank)
    ApplyBaseDesign sld
    AddHeader sld, "", titleText, badgeText

    AddCard sld, card1Title, card1Body, 70, 145, 395, 210, "primary"
    AddCard sld, card2Title, card2Body, 495, 145, 395, 210, "white"

    If Len(noteBoxText) > 0 Then
        AddNoteBox sld, noteBoxText, 70, 385, 820, 70
    End If

    AddSpeakerNotes sld, speakerNote
End Sub

Private Sub AddThreeCardSlide(ByVal pres As Presentation, ByVal titleText As String, ByVal badgeText As String, _
                              ByVal cardTitles As Variant, ByVal cardBodies As Variant, ByVal cardStyles As Variant, _
                              ByVal speakerNote As String)
    Dim sld As Slide
    Set sld = pres.Slides.Add(pres.Slides.Count + 1, ppLayoutBlank)
    ApplyBaseDesign sld
    AddHeader sld, "", titleText, badgeText

    Dim i As Long
    For i = 0 To 2
        AddNumberCard sld, CStr(i + 1), CStr(cardTitles(i)), CStr(cardBodies(i)), 70 + i * 285, 170, 260, 230, CStr(cardStyles(i))
    Next i

    AddSpeakerNotes sld, speakerNote
End Sub

Private Sub AddBulletSlide(ByVal pres As Presentation, ByVal titleText As String, ByVal badgeText As String, _
                           ByVal leadText As String, ByVal bullets As Variant, ByVal speakerNote As String)

    ApplyBaseDesign sld
    AddHeader sld, "", titleText, badgeText

    AddTextBox sld, leadText, 70, 145, 780, 80, 20, RGB(31, 41, 55), False
    AddBulletList sld, bullets, 95, 255, 720, 190, 21
    AddSpeakerNotes sld, speakerNote
End Sub

Private Sub AddFlowSlide(ByVal pres As Presentation, ByVal titleText As String, ByVal badgeText As String, _
                         ByVal stepTitles As Variant, ByVal stepBodies As Variant, ByVal speakerNote As String)
    Dim sld As Slide
    Set sld = pres.Slides.Add(pres.Slides.Count + 1, ppLayoutBlank)
    ApplyBaseDesign sld
    AddHeader sld, "", titleText, badgeText

    Dim i As Long, x As Single
    For i = 0 To 3
        x = 55 + i * 225
        AddFlowStep sld, CStr(i + 1), CStr(stepTitles(i)), CStr(stepBodies(i)), x, 190, 200, 190
        If i < 3 Then AddArrow sld, x + 200, 270
    Next i

    AddSpeakerNotes sld, speakerNote
End Sub

        .Cell(1, 3).Shape.TextFrame.TextRange.Text = "難しいこと"

        .Cell(2, 1).Shape.TextFrame.TextRange.Text = "HTMLアプリ"
        .Cell(2, 2).Shape.TextFrame.TextRange.Text = "閲覧、検索、画面上の入力、CSV・テキスト出力"
        .Cell(2, 3).Shape.TextFrame.TextRange.Text = "Word・Excelファイルへの直接書き込み"

        .Cell(3, 1).Shape.TextFrame.TextRange.Text = "データ保持"
        .Cell(3, 2).Shape.TextFrame.TextRange.Text = "同じ端末・同じブラウザ内での一時保存"
        .Cell(3, 3).Shape.TextFrame.TextRange.Text = "複数端末・複数職員での安定した共有"

        .Cell(4, 1).Shape.TextFrame.TextRange.Text = "PDF"
        .Cell(4, 2).Shape.TextFrame.TextRange.Text = "読み込み、画面上での書き込み、書き込み後の出力"

Private Sub AddTextBox(ByVal sld As Slide, ByVal txt As String, ByVal x As Single, ByVal y As Single, _
                       ByVal w As Single, ByVal h As Single, ByVal fontSize As Single, _
                       ByVal fontColor As Long, ByVal isBold As Boolean)
    Dim shp As Shape
    Set shp = sld.Shapes.AddTextbox(msoTextOrientationHorizontal, x, y, w, h)
    With shp.TextFrame
        .MarginLeft = 0
        .MarginRight = 0
        .MarginTop = 0
        .MarginBottom = 0
        .WordWrap = msoTrue
        .AutoSize = ppAutoSizeNone
        .TextRange.Text = txt
        With .TextRange.Font
            .NameFarEast = "Yu Gothic"
            .Name = "Yu Gothic"
            .Size = fontSize
            .Color.RGB = fontColor
            .Bold = IIf(isBold, msoTrue, msoFalse)
        End With
        .TextRange.ParagraphFormat.SpaceAfter = 4
    End With
End Sub

Private Sub FormatTable(ByVal tbl As Shape)
    Dim r As Long, c As Long

    For r = 1 To 4
        For c = 1 To 3
            With tbl.Table.Cell(r, c).Shape
                .Fill.ForeColor.RGB = IIf(r = 1, RGB(232, 242, 248), RGB(255, 255, 255))
                .Line.ForeColor.RGB = RGB(216, 224, 232)
                With .TextFrame.TextRange.Font
                    .NameFarEast = "Yu Gothic"
                    .Name = "Yu Gothic"
                    .Size = IIf(r = 1, 13.5, 12.5)
                    .Bold = IIf(r = 1 Or c = 1, msoTrue, msoFalse)
                    .Color.RGB = IIf(r = 1, RGB(23, 63, 95), RGB(31, 41, 55))
                End With
                .TextFrame.MarginLeft = 8
                .TextFrame.MarginRight = 8
                .TextFrame.MarginTop = 6
                .TextFrame.MarginBottom = 6
                .TextFrame.WordWrap = msoTrue
            End With
        Next c
    Next r
End Sub

Private Function StyleFill(ByVal styleName As String) As Long
    Select Case LCase(styleName)
        Case "primary"
            StyleFill = RGB(232, 242, 248)
        Case "accent"
            StyleFill = RGB(255, 244, 220)
        Case "green"
            StyleFill = RGB(232, 246, 239)
        Case "red"
            StyleFill = RGB(253, 236, 236)
        Case Else
            StyleFill = RGB(255, 255, 255)
    End Select
End Function
