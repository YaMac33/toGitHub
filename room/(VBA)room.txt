Option Explicit

' =========================================================
' reservations シート → reservations.js 出力（UTF-8版）
' 前提
' ・1行目 = 列名
' ・2行目以降 = データ
' ・A列も含めて全部読む
' ・1行目の最終列まで読む
' ・全列が空の行はスキップ
' ・1行目の途中空白列はエラー
' ・フラグ = "終日" のとき is_all_day = 1
' ・それ以外は is_all_day = 0
'
' 追加仕様
' ・「ＩＤ（システムＩＤ：自動発番）」は空欄必須
'   （値が入っていたらエラー）
' ・開始時刻 / 終了時刻 は
'   - 空欄OK
'   - 10:00 形式OK
'   - 10:00:00 形式OK
'   - Excelの時刻型OK
'   - 数値シリアルOK
' ・JS出力時の時刻は "HH:MM" に統一
' ・出力ファイルは UTF-8
'
' 出力形式
' window.APP_DATA = window.APP_DATA || {};
' window.APP_DATA.reservations = [ ... ];
' =========================================================

Public Sub ExportReservationsJs()
    On Error GoTo ErrHandler

    Dim ws As Worksheet
    Dim lastRow As Long
    Dim lastCol As Long
    Dim rowIndex As Long
    Dim colIndex As Long

    Dim baseFolder As String
    Dim outputPath As String

    Dim headers() As String
    Dim lineText As String
    Dim valueText As String
    Dim headerName As String

    Dim validationMessage As String
    Dim dataRowCount As Long
    Dim exportedCount As Long

    Dim content As String

    Set ws = ThisWorkbook.Worksheets("reservations")

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column

    If lastCol < 1 Then
        MsgBox "1行目に列名がありません。", vbExclamation
        Exit Sub
    End If

    If lastRow < 2 Then
        MsgBox "データ行がありません。2行目以降にデータを入れてください。", vbExclamation
        Exit Sub
    End If

    ReDim headers(1 To lastCol)

    ' 1行目: 列名取得
    For colIndex = 1 To lastCol
        headers(colIndex) = Trim(CStr(ws.Cells(1, colIndex).Value))
        If headers(colIndex) = "" Then
            MsgBox "1行目の列名に空欄があります。途中空白列は使えません。列 " & colIndex & " を確認してください。", vbExclamation
            Exit Sub
        End If
    Next colIndex

    ValidateRequiredHeaders headers

    validationMessage = ValidateRows(ws, headers, lastRow, lastCol)
    If validationMessage <> "" Then
        MsgBox validationMessage, vbExclamation
        Exit Sub
    End If

    dataRowCount = CountDataRows(ws, lastRow, lastCol)
    If dataRowCount = 0 Then
        MsgBox "出力対象のデータ行がありません。全て空白行の可能性があります。", vbExclamation
        Exit Sub
    End If

    baseFolder = ThisWorkbook.Path
    outputPath = baseFolder & Application.PathSeparator & "reservations.js"

    content = ""
    content = content & "window.APP_DATA = window.APP_DATA || {};" & vbCrLf
    content = content & "window.APP_DATA.reservations = [" & vbCrLf

    exportedCount = 0

    For rowIndex = 2 To lastRow
        If Not IsRowEmpty(ws, rowIndex, lastCol) Then
            exportedCount = exportedCount + 1

            content = content & "  {" & vbCrLf

            For colIndex = 1 To lastCol
                headerName = headers(colIndex)
                valueText = ToJsValue(headerName, ws.Cells(rowIndex, colIndex).Value)

                lineText = "    """ & EscapeJsString(headerName) & """: " & valueText & ","
                content = content & lineText & vbCrLf
            Next colIndex

            ' 追加変換項目
            content = content & "    ""is_all_day"": " & GetIsAllDayValue(ws, headers, rowIndex) & vbCrLf

            If exportedCount < dataRowCount Then
                content = content & "  }," & vbCrLf
            Else
                content = content & "  }" & vbCrLf
            End If
        End If
    Next rowIndex

    content = content & "];" & vbCrLf

    WriteTextUtf8 outputPath, content

    MsgBox "出力完了: " & outputPath, vbInformation
    Exit Sub

ErrHandler:
    MsgBox "エラーが発生しました: " & Err.Description, vbCritical
End Sub

Private Sub ValidateRequiredHeaders(ByRef headers() As String)
    Dim requiredHeaders As Variant
    Dim i As Long
    Dim found As Boolean
    Dim j As Long

    requiredHeaders = Array( _
        "ＩＤ（システムＩＤ：自動発番）", _
        "開始日", _
        "終了日", _
        "開始時刻", _
        "終了時刻", _
        "内容", _
        "予約種別", _
        "フラグ" _
    )

    For i = LBound(requiredHeaders) To UBound(requiredHeaders)
        found = False

        For j = LBound(headers) To UBound(headers)
            If headers(j) = requiredHeaders(i) Then
                found = True
                Exit For
            End If
        Next j

        If Not found Then
            Err.Raise vbObjectError + 1000, , "必須列が見つかりません: " & requiredHeaders(i)
        End If
    Next i
End Sub

Private Function ValidateRows(ByVal ws As Worksheet, ByRef headers() As String, ByVal lastRow As Long, ByVal lastCol As Long) As String
    Dim rowIndex As Long
    Dim msg As String

    msg = ""

    For rowIndex = 2 To lastRow
        If Not IsRowEmpty(ws, rowIndex, lastCol) Then
            msg = msg & ValidateRequiredCellsForRow(ws, headers, rowIndex)
            msg = msg & ValidateIdBlankForRow(ws, headers, rowIndex)
            msg = msg & ValidateDateRulesForRow(ws, headers, rowIndex)
            msg = msg & ValidateTimeRulesForRow(ws, headers, rowIndex)
        End If
    Next rowIndex

    ValidateRows = msg
End Function

Private Function ValidateRequiredCellsForRow(ByVal ws As Worksheet, ByRef headers() As String, ByVal rowIndex As Long) As String
    Dim requiredHeaders As Variant
    Dim i As Long
    Dim colIndex As Long
    Dim cellValue As String
    Dim msg As String

    requiredHeaders = Array( _
        "開始日", _
        "終了日", _
        "内容", _
        "予約種別" _
    )

    msg = ""

    For i = LBound(requiredHeaders) To UBound(requiredHeaders)
        colIndex = GetHeaderColumn(headers, CStr(requiredHeaders(i)))
        cellValue = Trim(CStr(ws.Cells(rowIndex, colIndex).Value))

        If cellValue = "" Then
            msg = msg & "行 " & rowIndex & " : " & requiredHeaders(i) & " が空欄です。" & vbCrLf
        End If
    Next i

    ValidateRequiredCellsForRow = msg
End Function

Private Function ValidateIdBlankForRow(ByVal ws As Worksheet, ByRef headers() As String, ByVal rowIndex As Long) As String
    Dim colIndex As Long
    Dim cellValue As String

    colIndex = GetHeaderColumn(headers, "ＩＤ（システムＩＤ：自動発番）")
    cellValue = Trim(CStr(ws.Cells(rowIndex, colIndex).Value))

    If cellValue <> "" Then
        ValidateIdBlankForRow = "行 " & rowIndex & " : ＩＤ（システムＩＤ：自動発番）は空欄にしてください。" & vbCrLf
    Else
        ValidateIdBlankForRow = ""
    End If
End Function

Private Function ValidateDateRulesForRow(ByVal ws As Worksheet, ByRef headers() As String, ByVal rowIndex As Long) As String
    Dim startDateText As String
    Dim endDateText As String
    Dim msg As String
    Dim startDateValue As Date
    Dim endDateValue As Date

    msg = ""

    startDateText = Trim(CStr(ws.Cells(rowIndex, GetHeaderColumn(headers, "開始日")).Value))
    endDateText = Trim(CStr(ws.Cells(rowIndex, GetHeaderColumn(headers, "終了日")).Value))

    If startDateText <> "" And Not IsDate(startDateText) Then
        msg = msg & "行 " & rowIndex & " : 開始日 が日付として不正です。" & vbCrLf
    End If

    If endDateText <> "" And Not IsDate(endDateText) Then
        msg = msg & "行 " & rowIndex & " : 終了日 が日付として不正です。" & vbCrLf
    End If

    If IsDate(startDateText) And IsDate(endDateText) Then
        startDateValue = CDate(startDateText)
        endDateValue = CDate(endDateText)

        If endDateValue < startDateValue Then
            msg = msg & "行 " & rowIndex & " : 終了日 が 開始日 より前です。" & vbCrLf
        End If
    End If

    ValidateDateRulesForRow = msg
End Function

Private Function ValidateTimeRulesForRow(ByVal ws As Worksheet, ByRef headers() As String, ByVal rowIndex As Long) As String
    Dim startTimeValue As Variant
    Dim endTimeValue As Variant
    Dim startTimeText As String
    Dim endTimeText As String
    Dim msg As String

    msg = ""

    startTimeValue = ws.Cells(rowIndex, GetHeaderColumn(headers, "開始時刻")).Value
    endTimeValue = ws.Cells(rowIndex, GetHeaderColumn(headers, "終了時刻")).Value

    startTimeText = Trim(CStr(startTimeValue))
    endTimeText = Trim(CStr(endTimeValue))

    If startTimeText <> "" Then
        If Not IsValidTimeValue(startTimeValue) Then
            msg = msg & "行 " & rowIndex & " : 開始時刻 は HH:MM または HH:MM:SS 形式で入力してください。" & vbCrLf
        End If
    End If

    If endTimeText <> "" Then
        If Not IsValidTimeValue(endTimeValue) Then
            msg = msg & "行 " & rowIndex & " : 終了時刻 は HH:MM または HH:MM:SS 形式で入力してください。" & vbCrLf
        End If
    End If

    ValidateTimeRulesForRow = msg
End Function

Private Function GetHeaderColumn(ByRef headers() As String, ByVal targetHeader As String) As Long
    Dim i As Long

    For i = LBound(headers) To UBound(headers)
        If headers(i) = targetHeader Then
            GetHeaderColumn = i
            Exit Function
        End If
    Next i

    Err.Raise vbObjectError + 1001, , "列が見つかりません: " & targetHeader
End Function

Private Function CountDataRows(ByVal ws As Worksheet, ByVal lastRow As Long, ByVal lastCol As Long) As Long
    Dim rowIndex As Long
    Dim countValue As Long

    countValue = 0

    For rowIndex = 2 To lastRow
        If Not IsRowEmpty(ws, rowIndex, lastCol) Then
            countValue = countValue + 1
        End If
    Next rowIndex

    CountDataRows = countValue
End Function

Private Function IsRowEmpty(ByVal ws As Worksheet, ByVal rowIndex As Long, ByVal lastCol As Long) As Boolean
    Dim colIndex As Long
    Dim cellText As String

    IsRowEmpty = True

    For colIndex = 1 To lastCol
        cellText = Trim(CStr(ws.Cells(rowIndex, colIndex).Value))
        If cellText <> "" Then
            IsRowEmpty = False
            Exit Function
        End If
    Next colIndex
End Function

Private Function ToJsValue(ByVal headerName As String, ByVal cellValue As Variant) As String
    Dim textValue As String

    textValue = Trim(CStr(cellValue))

    Select Case headerName
        Case "開始日", "終了日"
            If IsDate(cellValue) Then
                ToJsValue = """" & Format(CDate(cellValue), "yyyy-mm-dd") & """"
            Else
                ToJsValue = """" & EscapeJsString(NormalizeDateText(textValue)) & """"
            End If

        Case "開始時刻", "終了時刻"
            If Trim(CStr(cellValue)) = "" Then
                ToJsValue = """"""
            ElseIf IsValidTimeValue(cellValue) Then
                ToJsValue = """" & FormatTimeToHHMM(cellValue) & """"
            Else
                ToJsValue = """" & EscapeJsString(textValue) & """"
            End If

        Case Else
            ToJsValue = """" & EscapeJsString(textValue) & """"
    End Select
End Function

Private Function GetIsAllDayValue(ByVal ws As Worksheet, ByRef headers() As String, ByVal rowIndex As Long) As String
    Dim flagText As String

    flagText = Trim(CStr(ws.Cells(rowIndex, GetHeaderColumn(headers, "フラグ")).Value))

    If flagText = "終日" Then
        GetIsAllDayValue = "1"
    Else
        GetIsAllDayValue = "0"
    End If
End Function

Private Function EscapeJsString(ByVal textValue As String) As String
    Dim s As String

    s = textValue
    s = Replace(s, "\", "\\")
    s = Replace(s, """", "\""")
    s = Replace(s, vbCrLf, "\n")
    s = Replace(s, vbCr, "\n")
    s = Replace(s, vbLf, "\n")

    EscapeJsString = s
End Function

Private Function IsValidTimeValue(ByVal valueIn As Variant) As Boolean
    Dim s As String
    Dim parts() As String
    Dim hh As Long
    Dim mm As Long
    Dim ss As Long
    Dim n As Double

    If IsEmpty(valueIn) Then
        IsValidTimeValue = False
        Exit Function
    End If

    If IsNumeric(valueIn) Then
        n = CDbl(valueIn)
        If n >= 0 And n < 1 Then
            IsValidTimeValue = True
            Exit Function
        End If
    End If

    If IsDate(valueIn) Then
        IsValidTimeValue = True
        Exit Function
    End If

    s = Trim(CStr(valueIn))
    If s = "" Then
        IsValidTimeValue = False
        Exit Function
    End If

    parts = Split(s, ":")

    If UBound(parts) <> 1 And UBound(parts) <> 2 Then
        IsValidTimeValue = False
        Exit Function
    End If

    If Not IsNumeric(parts(0)) Or Not IsNumeric(parts(1)) Then
        IsValidTimeValue = False
        Exit Function
    End If

    hh = CLng(parts(0))
    mm = CLng(parts(1))

    If hh < 0 Or hh > 23 Then
        IsValidTimeValue = False
        Exit Function
    End If

    If mm < 0 Or mm > 59 Then
        IsValidTimeValue = False
        Exit Function
    End If

    If UBound(parts) = 2 Then
        If Not IsNumeric(parts(2)) Then
            IsValidTimeValue = False
            Exit Function
        End If

        ss = CLng(parts(2))
        If ss < 0 Or ss > 59 Then
            IsValidTimeValue = False
            Exit Function
        End If
    End If

    IsValidTimeValue = True
End Function

Private Function FormatTimeToHHMM(ByVal valueIn As Variant) As String
    Dim s As String
    Dim parts() As String
    Dim hh As Long
    Dim mm As Long
    Dim n As Double

    If IsNumeric(valueIn) Then
        n = CDbl(valueIn)
        If n >= 0 And n < 1 Then
            FormatTimeToHHMM = Format(CDate(n), "hh:mm")
            Exit Function
        End If
    End If

    If IsDate(valueIn) Then
        FormatTimeToHHMM = Format(CDate(valueIn), "hh:mm")
        Exit Function
    End If

    s = Trim(CStr(valueIn))
    parts = Split(s, ":")

    hh = CLng(parts(0))
    mm = CLng(parts(1))

    FormatTimeToHHMM = Format(hh, "00") & ":" & Format(mm, "00")
End Function

Private Function NormalizeDateText(ByVal valueText As String) As String
    NormalizeDateText = Replace(Trim(valueText), "/", "-")
End Function

Private Sub WriteTextUtf8(ByVal filePath As String, ByVal textContent As String)
    Dim stm As Object

    Set stm = CreateObject("ADODB.Stream")
    stm.Type = 2              ' adTypeText
    stm.Charset = "utf-8"
    stm.Open
    stm.WriteText textContent
    stm.SaveToFile filePath, 2 ' adSaveCreateOverWrite
    stm.Close

    Set stm = Nothing
End Sub
