cPeriodType).value) = "内" _ And CellText(ws.Cells(i, cMeetingTypeCode).value) = "HNKG" Then d = CellText(ws.Cells(i, cHeldDate).value) If d <> "" Then If maxDate = "" Or d > maxDate Then maxDate = d End If End If End If Next i FindSpecialTermEndDate = maxDate End Function Private Function HasMeetingRow( _ ByVal ws As Worksheet, _ ByVal rowIndex As Long, _ ByVal cMeetingName As Long, _ ByVal cHeldDate As Long _ ) As Boolean If CellText(ws.Cells(rowIndex, cMeetingName).value) = "" Then Exit Function If CellText(ws.Cells(rowIndex, cHeldDate).value) = "" Then Exit Function HasMeetingRow = True End Function Private Function IsSpecialCommitteeCode(ByVal code As String) As Boolean Select Case Trim$(code) Case "IKY", "TKY", "IKK", "TKK" IsSpecialCommitteeCode = True Case Else IsSpecialCommitteeCode = False End Select End Function Private Function NormalizeDurationMinutesText(ByVal value As Variant) As String If IsError(value) Then NormalizeDurationMinutesText = "" Exit Function End If If Trim$(CStr(value)) = "" Then NormalizeDurationMinutesText = "" ElseIf IsNumeric(value) Then If CDbl(value) > 0 And CDbl(value) < 1 Then NormalizeDurationMinutesText = CStr(Round(CDbl(value) * 1440, 0)) Else NormalizeDurationMinutesText = CStr(Round(CDbl(value), 0)) End If Else NormalizeDurationMinutesText = DigitsOnly(CStr(value)) End If End Function Private Function CellText(ByVal v As Variant) As String If IsError(v) Then CellText = "" Else CellText = Trim$(CStr(v)) End If End Function Private Function DigitsOnly(ByVal s As String) As String Dim i As Long Dim ch As String Dim r As String For i = 1 To Len(s) ch = Mid$(s, i, 1) If ch Like "#" Then r = r & ch Next i DigitsOnly = r End Function Private Function ToJsNumberOrNullText(ByVal s As String) As String If s = "" Then ToJsNumberOrNullText = "null" ElseIf IsNumeric(s) Then ToJsNumberOrNullText = s Else ToJsNumberOrNullText = "null" End If End Function Private Function EscapeJs(ByVal s As String) As String s = Replace(s, "\", "\\") s = Replace(s, """", "\""") s = Replace(s, vbCrLf, "\n") s = Replace(s, vbCr, "\n") s = Replace(s, vbLf, "\n") EscapeJs = s End Function Private Function FindHeaderColumn(ByVal ws As Worksheet, ByVal headerName As String) As Long Dim lastCol As Long Dim c As Long lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column For c = 1 To lastCol If vbDirectory) = "" Then MkDir folderPath End Sub Private Sub WriteTextFileUtf8(ByVal filePath As String, ByVal textData As String) Dim stm As Object Set stm = CreateObject("ADODB.Stream") stm.Type = 2 stm.Charset = "UTF-8" stm.Open stm.WriteText textData stm.SaveToFile filePath, 2 stm.Close Set stm = Nothing End Sub
